import java.io.BufferedReader;
import java.io.IOException;
import java.io.InputStreamReader;

public class Main {
	static BufferedReader bReader =new BufferedReader(new InputStreamReader(System.in));
    public static void main(String[] args) throws NumberFormatException, IOException {
    	String s[]=bReader.readLine().split(" ");
    	int n=Integer.parseInt(s[0]);//计划数
    	long t=Long.parseLong(s[1]);//总共天数
    	long d[]=new long[100010];//存天数
    	long b[]=new long[100010];//对应天数的草量
    	long ans=0;//答案
    	long sum = 0;//剩下多少草
    	d[0]=0;
    	b[0]=0;

    	d[n]=t+1;
    	b[n]=0;
    	for(int i=0;i<n;i++) {
    	        s = bReader.readLine().split(" ");

               d[i] = Long.parseLong(s[0]);
               b[i] = Long.parseLong(s[1]);
    	}
    	for(int i=0;i<n;i++) {
    		sum+=b[i];
    		if(sum>=d[i+1]-d[i]) {
    			ans+=d[i+1]-d[i];
    			sum-=d[i+1]-d[i];
    		}else {
				ans+=sum;
				sum=0;
			}
    	}
    	System.out.println(ans);

}
}
